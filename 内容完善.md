# 开机自启动服务

boa 开机自启动服务配置
```bash
sudo vim /etc/systemd/system/boa.service


[Unit]
Description=Boa Web Server
After=network.target syslog.target

[Service]
Type=simple
User=root
Group=root
ExecStart=/usr/sbin/boa -c /etc/boa
Restart=always
RestartSec=5
TimeoutSec=30
Environment="PATH=/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/sbin:/bin"

[Install]
WantedBy=multi-user.target
```

mjpg-streamer 开机自启动服务配置
```bash
sudo vim /etc/systemd/system/mjpg-streamer.service

[Unit]
Description=MJPEG Streamer
After=network.target

[Service]
Type=simple
User=kickpi
WorkingDirectory=/home/kickpi/mjpg-streamer/mjpg-streamer/mjpg-streamer-experimental
ExecStart=/home/kickpi/mjpg-streamer/mjpg-streamer/mjpg-streamer-experimental/mjpg_streamer -i "./input_uvc.so -d /dev/video25 -r 640x480 -f 30" -o "./output_http.so -p 8090 -w ./www"
Restart=always

[Install]
WantedBy=multi-user.target
```

权限配置
```bash
# 服务文件权限
sudo chmod 644 /etc/systemd/system/boa.service
sudo chmod 644 /etc/systemd/system/mjpg-streamer.service

# sudo usermod -aG video nobody
# sudo usermod -aG audio nobody

# 修复 mjpg-streamer 目录权限
# sudo chown -R nobody:nogroup /home/pi/mjpg-streamer
sudo chmod -R 755 /home/pi/mjpg-streamer/mjpg-streamer/mjpg-streamer-experimental
```

启动服务
```bash
# 重新加载 systemd 配置
sudo systemctl daemon-reload

# 启用开机自启
sudo systemctl enable boa.service
sudo systemctl enable mjpg-streamer.service

# 立即启动服务
sudo systemctl start boa.service
sudo systemctl start mjpg-streamer.service
```
